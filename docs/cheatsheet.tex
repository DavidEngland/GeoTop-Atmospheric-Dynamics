\sect{I. Differential Geometry}

\T{Objects on a physical manifold $\mathcal{M}$}
\begin{itemize}
  \item Physical manifold $\mathcal{M}$: smooth spatial domain (e.g., sphere)
  \item Tangent bundle $T\mathcal{M}$: holds velocity vector fields $u$
  \item Cotangent bundle $T^*\mathcal{M}$: dual; holds 1-form covectors $v$
\end{itemize}

\T{Graded algebra of $k$-forms $\Omega^k(\mathcal{M})$}
\begin{itemize}
  \item $0$-form $\zeta$: scalar field (relative vorticity, $p$, $\theta$)
  \item $1$-form $v$: velocity circulation d.o.f.
  \item $2$-form $h$: density/depth flux d.o.f.
\end{itemize}

\T{Key operators}\\[2pt]
\begin{tabular}{@{}l@{\;}l@{\;}p{1.4in}@{}}
\toprule
Op. & Action & Property \\
\midrule
$d$ & $\Omega^k\!\to\!\Omega^{k+1}$ & $d^2{=}0$ \\
$\wedge$ & $\Omega^p\!\times\!\Omega^q\!\to\!\Omega^{p+q}$ & $\alpha\wedge\beta{=}(-1)^{pq}\beta\wedge\alpha$ \\
$\star$ & $\Omega^k\!\to\!\Omega^{n-k}$ & metric-dependent \\
$\iota_u$ & $\Omega^k\!\to\!\Omega^{k-1}$ & contraction with $u$ \\
$\mathcal{L}_u$ & $\Omega^k\!\to\!\Omega^k$ & flow transport \\
\bottomrule
\end{tabular}

\T{Cartan's magic formula}
\[\mathcal{L}_u\alpha = d(\iota_u\alpha) + \iota_u(d\alpha)\]

\T{Stokes' theorem}
\[\int_{\partial M}\omega = \int_M d\omega\]

\sect{II. Geometric Fluid Dynamics}

\T{Vector-invariant incompressible Euler}
\[\partial_t v + \iota_u\omega + dB = 0,\quad \omega = dv\]
\[B = \tfrac{1}{2}|u|^2 + p + \Phi, \quad d(\star v) = 0\]
\[\partial_t\omega + \mathcal{L}_u\omega = 0\]

\T{Conserved invariants}\\[2pt]
\begin{tabular}{@{}ll@{}}
\toprule
Invariant & Expression \\
\midrule
Energy & $E = \tfrac{1}{2}\int_\Omega v\wedge\star v$ \\
Kelvin circ. & $\tfrac{d}{dt}\oint_{c(t)}v = 0$ \\
Helicity (3D) & $H = \int_\Omega v\wedge dv$ \\
Pot.\ vorticity & $q = (\omega+f)/\rho$ \\
Enstrophy (2D) & $Z = \tfrac{1}{2}\int\omega^2\,dA$ \\
\bottomrule
\end{tabular}

\sect{III. Discrete Exterior Calculus}

\T{Philosophy:} separate topology (combinatorial) from metric (geometric).

\begin{itemize}
  \item \T{Primal complex}: Delaunay triangles; $k$-cochains
  \item \T{Dual complex}: Voronoi polygons; $(n{-}k)$-cochains
  \item \T{Coboundary op.\ $D$}: discrete $d$; $D^2=0$ exactly
  \item \T{Discrete Hodge $M_k$}: diagonal mass matrix; sole error source
  \item \T{Whitney forms}: cochain-to-form interpolation
\end{itemize}

\T{Discrete energy identity}
\[\langle v,\,I_v(\tilde{D}_1 v)\rangle_1 = 0\]
Unconditional stability; no artificial dissipation.

\T{Applications:} MPAS, ICON, mimetic FD, compatible FE.

\sect{V. Lagrangian Coherent Structures}

\T{Space split:} LCS live on physical $\mathcal{M}$, while $M_0,M_\epsilon$ live in abstract state space.

\T{Flow map:} $F_{t_0}^t(x_0)$ — trajectory from $x_0$ at $t_0$.

\T{Cauchy--Green tensor}
\[C = \bigl(DF_{t_0}^t\bigr)^\top DF_{t_0}^t\]

\T{FTLE}
\[\sigma_{t_0}^t(x_0) = \frac{1}{|t-t_0|}\ln\sqrt{\lambda_{\max}(C)}\]

\T{LCS types}\\[2pt]
\begin{tabular}{@{}lll@{}}
\toprule
Type & FTLE & Geometry \\
\midrule
Repelling & fwd ridges & unstable mflds \\
Attracting & bwd ridges & stable mflds \\
Elliptic & closed lines & vortex cores \\
Parabolic & jet surfaces & shear barriers \\
\bottomrule
\end{tabular}

\T{Applications:} atmospheric rivers, hurricane eyewalls, jet streams, polar vortex edge, aerosol transport.

\sect{VI. Topological Data Analysis}

\T{Pipeline}
\[\text{cloud}\;\xrightarrow{\text{VR}}\;\text{filtration}\;\xrightarrow{H_k}\;\text{barcodes}\]

\begin{itemize}
  \item $\beta_0$: connected components
  \item $\beta_1$: loops / vortex rings
  \item $\beta_2$: enclosed voids
  \item Persistence diagram: birth--death pairs
  \item Barcode: feature lifetime
  \item Mapper: topological graph
\end{itemize}

\T{Applications:} convective organization, cloud morphology, turbulence intermittency, regime detection, radar analysis.
